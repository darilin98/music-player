21void UiController::beginRenderLoop()
22{
23 path_ = std::filesystem::current_path().string();
24
25 updateFileList();
26 nodelay(stdscr, TRUE);
27 ui_.renderFileList(files_, highlight_);
28 ui_.renderTrackQueue(track_queue_);
29 ui_.renderStatusBar(current_track_, playing_, player_.is_paused());
30
31 bool running = true;
32 while (running)
33 {
34 ui_.detectResize();
35 ui_.renderFileList(files_, highlight_);
36 ui_.renderTrackQueue(track_queue_);
37 ui_.renderStatusBar(current_track_, playing_, player_.is_paused());
38 ui_.updateAnimationFrame(current_track_, playing_, player_.is_paused());
39 napms(50);
40 if (!playing_ && !track_queue_.empty())
41 {
42 processNextTrackFromQueue();
43 }
44
45 switch (int pressed_key = getch()) {
46 case KEY_UP:
47 if (highlight_ > 0) highlight_--;
48 break;
49 case KEY_DOWN:
50 if (highlight_ < files_.size() - 1) highlight_++;
51 break;
52 case KEY_PLAY_TRACK:
53 processTrackSelection();
54 break;
55 case KEY_ADD_QUEUE:
56 addTrackToQueue();
57 break;
58 case KEY_PAUSE:
59 if (playing_) {
60 player_.is_paused() ? player_.resume_track() : player_.pause_track();
61 }
62 break;
63 case KEY_NEXT_QUEUE:
64 processNextTrackFromQueue();
65 break;
66 case KEY_VOLUME_UP:
67 player_.raise_volume();
68 break;
69 case KEY_VOLUME_DOWN:
70 player_.lower_volume();
71 break;
72 case KEY_QUIT:
73 stopTrackPlayback();
74 running = false;
75 break;
76 }
77 ui_.refreshAll();
78 }
79 endwin();
80}

82void UiController::beginTrackPlayback(const track_ptr_t& track)
83{
84 if (playback_thread_.joinable())
85 {
86 playback_thread_.join();
87 }
88 player_.load_track(track);
89 playing_ = true;
90 current_track_ = track;
91 playback_thread_ = std::thread([this]() {
92 player_.play_track();
93 playing_ = false;
94 current_track_ = nullptr;
95 ui_.renderStatusBar(current_track_, playing_, player_.is_paused());
96 });
97}

109 playback_thread_.join();

145void UiController::processTrackSelection()
146{
147 auto& selected = files_[highlight_];
148 std::filesystem::path selected_path = selected.path();
149
150 if (selected_path.filename() == "..") {
151 path_ = std::filesystem::path(path_).parent_path().string();
152 highlight_ = 0;
153 updateFileList();
154 } else if (is_directory(selected_path)) {
155 path_ = selected_path.string();
156 highlight_ = 0;
157 updateFileList();
158 } else
159 {
160 if (playing_)
161 stopTrackPlayback();
162 track_ptr_t track = dec_.decode_mp3(selected_path.string());
163 if (dynamic_cast<ErrorTrack*>(track.get()) != nullptr){
164 //Check type of returned track
165 showErrorPopup("Error opening file!");
166 return;
167 }
168 beginTrackPlayback(track);
169 }
170 ui_.renderStatusBar(current_track_, playing_, player_.is_paused());
171 ui_.renderFileList(files_, highlight_);
172}

187void UiController::processNextTrackFromQueue()
188{
189 if (playing_)
190 stopTrackPlayback();
191 if (!track_queue_.empty())
192 {
193 auto next_track = track_queue_.front();
194 track_queue_.pop_front();
195 beginTrackPlayback(next_track);
196 }
197 ui_.renderStatusBar(current_track_, playing_, player_.is_paused());
198 ui_.renderTrackQueue(track_queue_);
199}
